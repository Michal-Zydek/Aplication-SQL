CREATE TABLE UZYTKOWNIK
(
ID_UZYTKOWNIKA INT IDENTITY(1,1) PRIMARY KEY ,
IMIE VARCHAR(30) NOT NULL,
NAZWISKO VARCHAR(30) NOT NULL,
NR_INDEKSU INT NOT NULL,
EMAIL VARCHAR(40) NOT NULL,
TYP_KONTA CHAR(10) NOT NULL,
LOGIN VARCHAR(30) NOT NULL,
HASLO VARCHAR(200) NOT NULL
)

CREATE TABLE BAZA_DANYCH
(
ID_BAZA INT IDENTITY(1,1) PRIMARY KEY,
TYP CHAR(20) NOT NULL,
SCIEZKA VARCHAR(100) NOT NULL,
HASLO VARCHAR(30) NOT NULL,
LOGIN VARCHAR(30) NOT NULL,
NAZWA VARCHAR(20) NOT NULL,
ID_UZYTKOWNIKA INT REFERENCES UZYTKOWNIK(ID_UZYTKOWNIKA)
)

CREATE TABLE TEST
(
ID_TEST INT IDENTITY(1,1) PRIMARY KEY,
NAZWA VARCHAR(20) NOT NULL,
CZAS_START VARCHAR(20),
CZAS_STOP VARCHAR(20),
KLUCZ VARCHAR(10) NOT NULL,
ID_BAZA INT REFERENCES BAZA_DANYCH(ID_BAZA),
ID_UZYTKOWNIKA INT REFERENCES UZYTKOWNIK(ID_UZYTKOWNIKA)
)

CREATE TABLE PYTANIE
(
ID_PYTANIE INT IDENTITY(1,1) PRIMARY KEY,
PYTANIE VARCHAR(200) NOT NULL,
ID_TEST INT REFERENCES TEST(ID_TEST) NOT NULL
)

CREATE TABLE ODPOWIEDZ
(
ID_ODPOWIEDZI INT IDENTITY(1,1) PRIMARY KEY,
ODPOWIEDZ VARCHAR(200) NOT NULL,
ID_PYTANIE INT REFERENCES PYTANIE(ID_PYTANIE) NOT NULL,
ID_TEST INT REFERENCES TEST(ID_TEST) NOT NULL,
ID_UZYTKOWNIKA INT REFERENCES UZYTKOWNIK(ID_UZYTKOWNIKA)NOT NULL
)
